def formatlines(left_gutter = "Line ")
        all_lines = split("\n")
        max_width = all_lines.size.to_s.size
       all_lines.each.with_index do |line,index|
                string_format = "#{left_gutter} %0#{max_width}d: " % (index + 1).to_s
                all_lines[index] =  string_format + line
        end
end
